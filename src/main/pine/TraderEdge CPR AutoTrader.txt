// ─── INPUTS ───────────────────────────────────────────────
tradeSymbol = input.string("NSE:NIFTY26MARFUT", "Broker Symbol ", group="▶ SIGNAL SETTINGS",  tooltip="The trading symbol for order alerts")
tradeQuantity = input.int(65, "Quantity", group="▶ SIGNAL SETTINGS" , tooltip="The quantity to be traded. For futures this should be in multiples of lot size")

maxBreakDistance = input.float(25.0, "Breakout Buffer", step=1, group="▶ SIGNAL SETTINGS", tooltip="The maximum points the breakout candle can be away from the pivot level. If it exceeds this level alerts are ignored")
slBufferPoints = input.float(15.0, "SL Buffer", step=1, group="▶ SIGNAL SETTINGS",  tooltip="The additional stop loss buffer points added to pivot levels stop losses. ")

r2s2SlBuffer = input.float(25.0, "R2/S2 SL Points", step=1, group="▶ SIGNAL SETTINGS", tooltip="The stop loss points for single level pivots like R2/R3/R4 and S2/S3/S4. SL Buffer is also added / reduced to this.")
maxSlPoints = input.float(100.0, "Max SL Points", step=1, group="▶ SIGNAL SETTINGS", tooltip="The maximum stop loss points beyond which alerts are ignored. This is calculated from the breakout candle close levels to pivot based stop loss levels")

showEventLog = input.bool(true, "Show Event Log", group="▶ EVENT LOGS", inline="SIGNALS3")

// ─── PREVIOUS DAY DATA ───────────────────────────────────
ph = request.security(syminfo.tickerid, "D", high[1],  lookahead=barmerge.lookahead_on)
pl = request.security(syminfo.tickerid, "D", low[1],   lookahead=barmerge.lookahead_on)
pc = request.security(syminfo.tickerid, "D", close[1], lookahead=barmerge.lookahead_on)

// ─── PREVIOUS WEEK DATA ───────────────────────────────────
wh = request.security(syminfo.tickerid, "W", high[1],  lookahead=barmerge.lookahead_on)
wl = request.security(syminfo.tickerid, "W", low[1],   lookahead=barmerge.lookahead_on)
wc = request.security(syminfo.tickerid, "W", close[1], lookahead=barmerge.lookahead_on)


// ─── DAILY CPR CALCULATION ─────────────────────────────────
pp = (ph + pl + pc) / 3
bc = (ph + pl) / 2
tc = pp * 2 - bc

r1 = pp * 2 - pl
s1 = pp * 2 - ph

cprRange = ph - pl

r2 = pp + cprRange
s2 = pp - cprRange

r3 = ph + 2 * (pp - pl)
s3 = pl - 2 * (ph - pp)

r4 = r3 + (r2 - r1)
s4 = s3 - (s1 - s2)

dTop = math.max(pp, math.max(bc, tc))
dBot = math.min(pp, math.min(bc, tc))

dailyVeryBull = close > r1 and close > ph
dailyVeryBear = close < s1 and close < pl

dailyBull = close > dTop
dailyBear = close < dBot


// ─── WEEKLY CPR (for trend determination) ───────────────────
wpp = (wh + wl + wc) / 3
wbc = (wh + wl) / 2
wtc = wpp * 2 - wbc

wTop = math.max(wpp, math.max(wbc, wtc))
wBot = math.min(wpp, math.min(wbc, wtc))

weeklyBull = close > wTop
weeklyBear = close < wBot

wr1 = 2 * wpp - wl
ws1 = 2 * wpp - wh

weeklyVeryBull = close > wr1 and close > wh
weeklyVeryBear = close < ws1 and close < wl


//────────────────────────────────────────────
// Trend text
//────────────────────────────────────────────
dailyTrend =
     dailyVeryBull ? "Strong Bullish" :
     dailyVeryBear ? "Strong Bearish" :
     dailyBull     ? "Bullish" :
     dailyBear     ? "Bearish" :
     "Neutral"

weeklyTrend =
     weeklyVeryBull ? "Strong Bullish" :
     weeklyVeryBear ? "Strong Bearish" :
     weeklyBull     ? "Bullish" :
     weeklyBear     ? "Bearish" :
     "Neutral"


//────────────────────────────
// Recommendation rules
//────────────────────────────
buyNearCpr =
     (weeklyTrend == "Bullish" or weeklyTrend == "Strong Bullish") and
     (dailyTrend == "Bullish")

buyNearR1PDH =
     (weeklyTrend == "Bullish" or weeklyTrend == "Strong Bullish") and
     (dailyTrend == "Strong Bullish")

     (weeklyTrend == "Bullish" or weeklyTrend == "Strong Bullish") and
     (dailyTrend == "Bearish")

sellNearCpr =
     (weeklyTrend == "Bearish" or weeklyTrend == "Strong Bearish") and
     (dailyTrend == "Bearish")

sellNearS1PDL =
     (weeklyTrend == "Bearish" or weeklyTrend == "Strong Bearish") and
     (dailyTrend == "Strong Bearish")

sellReversal =
     (weeklyTrend == "Bearish" or weeklyTrend == "Strong Bearish") and
     (dailyTrend == "Bullish")

isBuyRecommendation = buyNearCpr or buyReversal or buyNearR1PDH
isSellRecommendation = sellNearCpr or sellReversal or sellNearS1PDL

// Label history management

//────────────────────────────

//────────────────────────────

rawBuyBreakR1PDH = isBuyRecommendation and barstate.isconfirmed and (close > r1 and close > ph) and (open < r1 or open < ph or low < r1 or low < ph) and (close > open)
rawBuyBreakR2 = isBuyRecommendation and barstate.isconfirmed and (close > r2) and (open < r2 or low < r2) and (close > open)
rawBuyBreakR3 = isBuyRecommendation and barstate.isconfirmed and (close > r3) and (open < r3 or low < r3) and (close > open)
rawBuyBreakR4 = isBuyRecommendation and barstate.isconfirmed and (close > r4) and (open < r4 or low < r4) and (close > open)
rawBuyBreakS1PDL = isBuyRecommendation and barstate.isconfirmed and (close > s1 and close > pl) and (open < s1 or open < pl or low < s1 or low < pl) and (close > open)

// Distance checks

validBuyBreakCpr = distanceFromCPR <= maxBreakDistance
distanceFromR1PDH = close - math.max(r1, ph)
validBuyBreakR1PDH = distanceFromR1PDH <= maxBreakDistance
distanceFromR2 = close - r2
validBuyBreakR2 = distanceFromR2 <= maxBreakDistance
distanceFromR3 = close - r3
validBuyBreakR3 = distanceFromR3 <= maxBreakDistance
distanceFromR4 = close - r4
validBuyBreakR4 = distanceFromR4 <= maxBreakDistance
distanceFromS1PDL = close - math.max(s1, pl)
validBuyBreakS1PDL = distanceFromS1PDL <= maxBreakDistance

// SL distance checks
slDistBuyCpr = close - (math.min(tc, bc) - slBufferPoints)
validSlBuyCpr = slDistBuyCpr <= maxSlPoints
slDistBuyR1PDH = close - (math.min(r1, ph) - slBufferPoints)
validSlBuyR1PDH = slDistBuyR1PDH <= maxSlPoints

validSlBuyR2 = slDistBuyR2 <= maxSlPoints
slDistBuyR3 = close - (r3 - r2s2SlBuffer)
validSlBuyR3 = slDistBuyR3 <= maxSlPoints
slDistBuyR4 = close - (r4 - r2s2SlBuffer)
validSlBuyR4 = slDistBuyR4 <= maxSlPoints
slDistBuyS1PDL = close - (math.min(s1, pl) - slBufferPoints)
validSlBuyS1PDL = slDistBuyS1PDL <= maxSlPoints

// Final buy signals
buyBreakCpr = rawBuyBreakCpr and validBuyBreakCpr and validSlBuyCpr

buyBreakR2 = rawBuyBreakR2 and validBuyBreakR2 and validSlBuyR2

slDistSellS4 = (s4 + r2s2SlBuffer) - close
validSlSellS4 = slDistSellS4 <= maxSlPoints

validSlSellR1PDH = slDistSellR1PDH <= maxSlPoints

// Final sell signals
